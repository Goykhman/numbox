deleted_node_base_ctor_error = "Intended for inheritance."

class NodeBase(StructRefProxy):
    def __new__(cls, name):
        raise NotImplementedError(deleted_node_base_ctor_error)

    @property
    @njit(**default_jit_options)
    def name(self):
        return self.name

    def __str__(self):
        return self.name
